    101:['ravi','python',6000],
}

let courseNotFound = new Error("Given course not Existed")
let InvalidCourseFee = new Error("Course Fee More than actual fee");

var sname = prompt("Enter Student Name :")
var course = prompt("Enter Student Course :")
var fee = Number(prompt("Enter Student Fee :"))

if(course in courses){
    if(fee > courses[course]){
        throw InvalidCourseFee
    } else {
        students[102] = [sname,course,fee]
    }
} else {
    throw courseNotFound
